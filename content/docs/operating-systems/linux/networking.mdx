## 2. Network Configuration

<div className="p-6 bg-white dark:bg-gray-800 rounded-xl shadow-sm mb-8">
  <Tabs items={["Debian/Ubuntu", "Red Hat/CentOS", "Arch Linux"]}>
    <Tab>
      <div className="p-4 bg-gray-50 dark:bg-gray-900 rounded-lg">
        <h4 className="font-semibold text-gray-800 dark:text-white mb-3">Debian-based Systems</h4>
        ```bash
        # Network interface configuration
        sudo nano /etc/network/interfaces

        # Example configuration
        auto eth0
        iface eth0 inet static
            address 192.168.1.100
            netmask 255.255.255.0
            gateway 192.168.1.1
            dns-nameservers 8.8.8.8 8.8.4.4
        
        # Restart networking
        sudo systemctl restart networking
        ```
      </div>
    </Tab>
    <Tab>
      <div className="p-4 bg-gray-50 dark:bg-gray-900 rounded-lg">
        <h4 className="font-semibold text-gray-800 dark:text-white mb-3">Red Hat-based Systems</h4>
        ```bash
        # Network interface configuration
        sudo nano /etc/sysconfig/network-scripts/ifcfg-eth0

        # Example configuration
        DEVICE=eth0
        BOOTPROTO=static
        IPADDR=192.168.1.100
        NETMASK=255.255.255.0
        GATEWAY=192.168.1.1
        DNS1=8.8.8.8
        DNS2=8.8.4.4
        ONBOOT=yes
        
        # Restart networking
        sudo systemctl restart NetworkManager
        ```
      </div>
    </Tab>
    <Tab>
      <div className="p-4 bg-gray-50 dark:bg-gray-900 rounded-lg">
        <h4 className="font-semibold text-gray-800 dark:text-white mb-3">Arch Linux</h4>
        ```bash
        # Using systemd-networkd
        sudo nano /etc/systemd/network/20-wired.network

        # Example configuration
        [Match]
        Name=eth0

        [Network]
        Address=192.168.1.100/24
        Gateway=192.168.1.1
        DNS=8.8.8.8
        DNS=8.8.4.4
        
        # Restart networking
        sudo systemctl restart systemd-networkd
        ```
      </div>
    </Tab>
  </Tabs>
</div>

## 3. Essential Networking Commands

<div className="p-6 bg-white dark:bg-gray-800 rounded-xl shadow-sm mb-8">
  <div className="grid grid-cols-1 md:grid-cols-2 gap-6">
    <div className="p-4 bg-green-50 dark:bg-green-900/30 rounded-lg">
      <h4 className="font-semibold text-green-700 dark:text-green-300 mb-2">Basic Commands</h4>
      <div className="space-y-2">
        ```bash
        # Show IP addresses
        ip addr show
        
        # Show routing table
        ip route show
        
        # Test connectivity
        ping example.com
        
        # Trace route to host
        traceroute example.com
        
        # Show network statistics
        netstat -tuln
        
        # DNS lookup
        dig example.com
        ```
      </div>
    </div>
    <div className="p-4 bg-red-50 dark:bg-red-900/30 rounded-lg">
      <h4 className="font-semibold text-red-700 dark:text-red-300 mb-2">Advanced Commands</h4>
      <div className="space-y-2">
        ```bash
        # Capture network traffic
        tcpdump -i eth0
        
        # Show socket statistics
        ss -tuln
        
        # Network interface statistics
        ip -s link show
        
        # ARP table
        ip neigh show
        
        # Network scanning
        nmap -sn 192.168.1.0/24
        ```
      </div>
    </div>
  </div>
</div>

## 4. Network Services

<div className="p-6 bg-white dark:bg-gray-800 rounded-xl shadow-sm mb-8">
  <div className="space-y-6">

      <ul className="list-disc list-inside text-gray-600 dark:text-gray-300">
        <li>DHCP (Dynamic Host Configuration Protocol)</li>
        <li>DNS (Domain Name System)</li>
        <li>SSH (Secure Shell)</li>
        <li>FTP (File Transfer Protocol)</li>
        <li>HTTP/HTTPS (Web Services)</li>
      </ul>
    </div>

    <Tabs items={["DHCP Configuration", "DNS Setup", "SSH Configuration"]}>
      <Tab>
        <div className="p-4 bg-gray-50 dark:bg-gray-900 rounded-lg">
          <h4 className="font-semibold text-gray-800 dark:text-white mb-3">DHCP Server Setup</h4>
          ```bash
          # Install DHCP server
          sudo apt install isc-dhcp-server   # Debian/Ubuntu
          sudo dnf install dhcp-server       # Red Hat/CentOS
          sudo pacman -S dhcp                # Arch Linux

          # Basic configuration
          sudo nano /etc/dhcp/dhcpd.conf

          subnet 192.168.1.0 netmask 255.255.255.0 {
              range 192.168.1.100 192.168.1.200;
              option routers 192.168.1.1;
              option domain-name-servers 8.8.8.8, 8.8.4.4;
          }
          ```
        </div>
      </Tab>
      <Tab>
        <div className="p-4 bg-gray-50 dark:bg-gray-900 rounded-lg">
          <h4 className="font-semibold text-gray-800 dark:text-white mb-3">DNS Server Setup</h4>
          ```bash
          # Install BIND DNS server
          sudo apt install bind9              # Debian/Ubuntu
          sudo dnf install bind              # Red Hat/CentOS
          sudo pacman -S bind                # Arch Linux

          # Basic zone configuration
          sudo nano /etc/bind/named.conf.local

          zone "example.com" {
              type master;
              file "/etc/bind/zones/example.com.zone";
          };
          ```
        </div>
      </Tab>
      <Tab>
        <div className="p-4 bg-gray-50 dark:bg-gray-900 rounded-lg">
          <h4 className="font-semibold text-gray-800 dark:text-white mb-3">SSH Server Setup</h4>
          ```bash
          # Install OpenSSH server
          sudo apt install openssh-server     # Debian/Ubuntu
          sudo dnf install openssh-server     # Red Hat/CentOS
          sudo pacman -S openssh              # Arch Linux

          # Basic configuration
          sudo nano /etc/ssh/sshd_config

          Port 22
          PermitRootLogin no
          PasswordAuthentication yes
          PubkeyAuthentication yes
          ```
        </div>
      </Tab>
    </Tabs>
  </div>
</div>

## 5. Network Security

<div className="p-6 bg-white dark:bg-gray-800 rounded-xl shadow-sm mb-8">
  <div className="p-4 bg-orange-50 dark:bg-orange-900/30 rounded-lg w-full">
    <h4 className="font-semibold text-orange-700 dark:text-orange-300 mb-2">Firewall Configuration</h4>
    ```bash
    # UFW (Uncomplicated Firewall)
    sudo ufw enable
    sudo ufw allow 22/tcp
    sudo ufw status
    
    # firewalld
    sudo firewall-cmd --permanent --add-service=ssh
    sudo firewall-cmd --reload
    
    # iptables
    sudo iptables -A INPUT -p tcp --dport 22 -j ACCEPT
    sudo iptables-save
    ```
  </div>
</div>
