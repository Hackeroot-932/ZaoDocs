## 2. Git Basics

<Tabs items={["Repository Operations", "Making Changes", "History & Undoing", "Remote Operations"]}>
  <Tab>
    <div className="p-4 border rounded-lg bg-white dark:bg-gray-800 my-4">

      <Accordions>
        <Accordion title="Creating Repositories">
          ```bash
          # Initialize a new repository
          git init [project-name]

          # Clone existing repository
          git clone [url]
          ```
        </Accordion>
        <Accordion title="Basic Commands">
          ```bash
          # Check repository status
          git status

          # Add files to staging
          git add [file]
          git add .  # Add all files

          # Commit changes
          git commit -m "commit message"
          ```
        </Accordion>
      </Accordions>
    </div>
  </Tab>
  <Tab>
    <div className="p-4 border rounded-lg bg-white dark:bg-gray-800 my-4">
      <h4 className="font-bold mb-2 text-green-600 dark:text-green-400">Making Changes</h4>
      
      <Accordions>
        <Accordion title="Staging Changes">
          ```bash
          # View changes
          git diff
          git diff --staged

          # Stage files
          git add [file]
          git add -p  # Interactive staging

          # Remove files
          git rm [file]
          git rm --cached [file]  # Untrack file
          ```
        </Accordion>
        <Accordion title="Committing Changes">
          ```bash
          # Basic commit
          git commit -m "message"

          # Amend last commit
          git commit --amend

          # Interactive rebase
          git rebase -i HEAD~[number]
          ```
        </Accordion>
      </Accordions>
    </div>
  </Tab>
  <Tab>
    <div className="p-4 border rounded-lg bg-white dark:bg-gray-800 my-4">
      <h4 className="font-bold mb-2 text-purple-600 dark:text-purple-400">History & Undoing</h4>
      
      <Accordions>
        <Accordion title="Viewing History">
          ```bash
          # View commit history
          git log
          git log --oneline
          git log --graph
          git log -p  # Show patches
          ```
        </Accordion>
        <Accordion title="Undoing Changes">
          ```bash
          # Discard changes in working directory
          git checkout -- [file]

          # Reset staging area
          git reset HEAD [file]

          # Revert a commit
          git revert [commit]

          # Reset to specific commit
          git reset --hard [commit]
          ```
        </Accordion>
      </Accordions>
    </div>
  </Tab>
  <Tab>
    <div className="p-4 border rounded-lg bg-white dark:bg-gray-800 my-4">
      <h4 className="font-bold mb-2 text-red-600 dark:text-red-400">Remote Operations</h4>
      
      <Accordions>
        <Accordion title="Managing Remotes">
          ```bash
          # Add remote
          git remote add [name] [url]

          # View remotes
          git remote -v

          # Remove remote
          git remote remove [name]
          ```
        </Accordion>
        <Accordion title="Syncing Changes">
          ```bash
          # Fetch changes
          git fetch [remote]

          # Pull changes
          git pull [remote] [branch]

          # Push changes
          git push [remote] [branch]
          ```
        </Accordion>
      </Accordions>
    </div>
  </Tab>

## 3. Git Branching

<Tabs items={["Branch Basics", "Merging", "Remote Branches", "Workflows"]}>

      ```bash
      # List branches
      git branch

      # Create branch
      git branch [branch-name]

      # Switch branch
      git checkout [branch-name]
      git switch [branch-name]  # New command

      # Create and switch
      git checkout -b [branch-name]
      git switch -c [branch-name]  # New command
      ```
    </div>
  </Tab>
  <Tab>
    <div className="p-4 border rounded-lg bg-white dark:bg-gray-800 my-4">
      <h4 className="font-bold mb-2 text-green-600 dark:text-green-400">Merging</h4>
      
      ```bash
      # Merge branch
      git merge [branch]

      # Abort merge
      git merge --abort

      # Rebase
      git rebase [branch]
      git rebase -i HEAD~[number]  # Interactive
      ```
    </div>
  </Tab>
  <Tab>

      ```bash
      # Track remote branch
      ```

          ```bash
          # By commit SHA

          # By reference
          git show HEAD^  # Parent of HEAD

          ```
        </Accordion>
        <Accordion title="Interactive Staging">
          ```bash
          # Interactive add
          git add -i

          # Interactive patch
          git add -p

          ```
        </Accordion>
      </Accordions>

  <Tab>

      <Accordions>
        <Accordion title="Binary Search">
          ```bash
          # Start bisect
          git bisect start

          # Mark current version as bad
          git bisect bad

          # Mark last known good version
          git bisect good v1.0

          # Finish bisect
          git bisect reset
          ```
        </Accordion>
        <Accordion title="Blame">
          ```bash
          # View file blame
          git blame [file]

          # Ignore whitespace
          git blame -w [file]

          # Show specific revision
          git blame v1.0 [file]
          ```
        </Accordion>
      </Accordions>
    </div>
  </Tab>
  <Tab>
    <div className="p-4 border rounded-lg bg-white dark:bg-gray-800 my-4">

      <Accordions>
        <Accordion title="Rewriting History">

          ```bash
          # Change last commit
          git commit --amend

          # Rebase interactively
          git rebase -i HEAD~3

          # Force push changes
          git push --force-with-lease
          ```
        </Accordion>
        <Accordion title="Reset & Clean">

          ```bash
          # Reset staging area
          git reset HEAD

          # Reset working directory
          ```

        </Accordion>

          ```bash
          git pull --recurse-submodules

          # Execute command in each submodule
          git submodule foreach 'git pull origin master'
          ```
        </Accordion>
