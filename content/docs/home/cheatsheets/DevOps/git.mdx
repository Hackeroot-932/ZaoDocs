## 2. Git Basics

<Tabs items={["Repository Operations", "Making Changes", "History & Undoing", "Remote Operations"]}>
  <Tab>
    <div className="p-4 border rounded-lg bg-white dark:bg-gray-800 my-4">
      <h4 className="font-bold mb-2 text-blue-600 dark:text-blue-400">Repository Operations</h4>
      <div className="space-y-6">
        <div>
          <h5 className="font-semibold mb-2">Creating Repositories</h5>
          ```bash
          # Initialize a new repository
          git init [project-name]

          # Clone existing repository
          git clone [url]
          ```
        </div>
        <div>
          <h5 className="font-semibold mb-2">Basic Commands</h5>
          ```bash
          # Check repository status
          git status

          # Add files to staging
          git add [file]
          git add .  # Add all files

          # Commit changes
          git commit -m "commit message"
          ```
        </div>
      </div>
    </div>
  </Tab>
  <Tab>
    <div className="p-4 border rounded-lg bg-white dark:bg-gray-800 my-4">
      <h4 className="font-bold mb-2 text-green-600 dark:text-green-400">Making Changes</h4>
      <div className="space-y-6">
        <div>
          <h5 className="font-semibold mb-2">Staging Changes</h5>
          ```bash
          # View changes
          git diff
          git diff --staged

          # Stage files
          git add [file]
          git add -p  # Interactive staging

          # Remove files
          git rm [file]
          git rm --cached [file]  # Untrack file
          ```
        </div>
        <div>
          <h5 className="font-semibold mb-2">Committing Changes</h5>
          ```bash
          # Basic commit
          git commit -m "message"

          # Amend last commit
          git commit --amend

          # Interactive rebase
          git rebase -i HEAD~[number]
          ```
        </div>
      </div>
    </div>
  </Tab>
  <Tab>
    <div className="p-4 border rounded-lg bg-white dark:bg-gray-800 my-4">
      <h4 className="font-bold mb-2 text-purple-600 dark:text-blue-400">History & Undoing</h4>
      <div className="space-y-6">
        <div>
          <h5 className="font-semibold mb-2">Viewing History</h5>
          ```bash
          # View commit history
          git log
          git log --oneline
          git log --graph
          git log -p  # Show patches
          ```
        </div>
        <div>
          <h5 className="font-semibold mb-2">Undoing Changes</h5>
          ```bash
          # Discard changes in working directory
          git checkout -- [file]

          # Reset staging area
          git reset HEAD [file]

          # Revert a commit
          git revert [commit]

          # Reset to specific commit
          git reset --hard [commit]
          ```
        </div>
      </div>
    </div>
  </Tab>
  <Tab>
    <div className="p-4 border rounded-lg bg-white dark:bg-gray-800 my-4">
      <h4 className="font-bold mb-2 text-red-600 dark:text-red-400">Remote Operations</h4>
      <div className="space-y-6">
        <div>
          <h5 className="font-semibold mb-2">Managing Remotes</h5>
          ```bash
          # Add remote
          git remote add [name] [url]

          # View remotes
          git remote -v

          # Remove remote
          git remote remove [name]
          ```
        </div>
        <div>
          <h5 className="font-semibold mb-2">Syncing Changes</h5>
          ```bash
          # Fetch changes
          git fetch [remote]

          # Pull changes
          git pull [remote] [branch]

          # Push changes
          git push [remote] [branch]
          ```
        </div>
      </div>
    </div>
  </Tab>
</Tabs>

          ```bash
          # By commit SHA
          git show a1b2c3d4

          # By branch
          git show branch-name

          # By date
          git show master@{yesterday}

          # By reference
          git show HEAD^  # Parent of HEAD
          git show HEAD~3  # Great-grandparent of HEAD
          ```
        </div>
        <div>
          <h5 className="font-semibold mb-2">Interactive Staging</h5>
          ```bash
          # Interactive add
          git add -i

          # Interactive patch
          git add -p

          # Interactive rebase
          git rebase -i HEAD~3
          ```
        </div>
      </div>
    </div>
  </Tab>
  <Tab>
    <div className="p-4 border rounded-lg bg-white dark:bg-gray-800 my-4">

      <div className="space-y-6">
        <div>
          <h5 className="font-semibold mb-2">Binary Search</h5>
          ```bash
          # Start bisect
          git bisect start

          # Mark current version as bad
          git bisect bad

          # Mark last known good version
          git bisect good v1.0

          # Finish bisect
          git bisect reset
          ```
        </div>
        <div>
          <h5 className="font-semibold mb-2">Blame</h5>
          ```bash
          # View file blame
          git blame [file]

          # Ignore whitespace
          git blame -w [file]

          # Show specific revision
          git blame v1.0 [file]
          ```
        </div>
      </div>
    </div>
  </Tab>
  <Tab>
    <div className="p-4 border rounded-lg bg-white dark:bg-gray-800 my-4">
      <h4 className="font-bold mb-2 text-purple-600 dark:text-blue-400">History Management</h4>
      <div className="space-y-6">
        <div>
          <h5 className="font-semibold mb-2">Rewriting History</h5>
          ```bash
          # Change last commit
          git commit --amend

          # Rebase interactively
          git rebase -i HEAD~3

          # Force push changes
          git push --force-with-lease
          ```
        </div>
        <div>
          <h5 className="font-semibold mb-2">Reset & Clean</h5>
          ```bash
          # Reset staging area
          git reset HEAD

          # Reset working directory
          git checkout -- .

          # Clean untracked files
          git clean -fd
          ```
        </div>
      </div>
    </div>
  </Tab>

          ```bash
          # Add submodule
          git submodule add [url] [path]

          # Initialize submodules
          git submodule init

          # Update submodules
          git submodule update

          # Update all submodules
          git submodule update --init --recursive
          ```
        </div>

          ```bash
          # Clone with submodules
          git clone --recursive [url]

          # Pull all submodules
          git pull --recurse-submodules

          # Execute command in each submodule
          git submodule foreach 'git pull origin master'
          ```
